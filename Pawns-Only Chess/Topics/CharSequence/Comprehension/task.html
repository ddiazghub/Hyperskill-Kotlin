Comprehension

Look at the following code:

fun main() {

    val input = "abracadabra"
    val count = countCharacterA(input)
    println("$count")
}

fun countCharacterA(charSeq: CharSequence): Int {
    var count = 0
    for (element in charSeq) {
        if (element == 'a') {
            count++
        }
    }
    return count
}
